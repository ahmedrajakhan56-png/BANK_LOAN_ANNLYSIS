```python
"""
BANK LOAN ANALYSIS - PYTHON ETL
Business Goal: Clean 38,576 loan records, calculate key financial metrics
Focus: Data quality ensures accurate lending decisions
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

# ============================================
# 1. LOAD DATA - 38,576 loan applications
# ============================================
print("🚀 Loading bank loan data...")
df = pd.read_csv('financial_loan_data.csv')
print(f"📊 Dataset Shape: {df.shape}")  # (38,576, 24)
print(f"📅 Data Period: {df['issue_date'].min()} to {df['issue_date'].max()}")

# ============================================
# 2. DATA CLEANING - Because dirty data = wrong lending decisions
# ============================================
print("\n🧹 Cleaning data for accurate business insights...")

# Check for missing values
print(f"❓ Missing values before cleaning:")
print(df.isnull().sum())

# Handle missing values
# For numeric columns: fill with median
numeric_cols = ['annual_income', 'dti', 'installment', 'int_rate', 'loan_amount']
for col in numeric_cols:
    df[col].fillna(df[col].median(), inplace=True)

# For categorical: fill with mode
categorical_cols = ['grade', 'sub_grade', 'home_ownership', 'verification_status', 'purpose']
for col in categorical_cols:
    df[col].fillna(df[col].mode()[0], inplace=True)

# Remove duplicates
initial_count = len(df)
df.drop_duplicates(subset=['id'], inplace=True)
dupes_removed = initial_count - len(df)
print(f"✅ Removed {dupes_removed} duplicate applications")

# Remove outliers (extreme values that don't make business sense)
df = df[df['loan_amount'] > 0]  # Loan amount must be positive
df = df[df['loan_amount'] < 50000]  # Remove outliers > $50K
df = df[df['annual_income'] < 1000000]  # Remove unrealistic incomes

# ============================================
# 3. FEATURE ENGINEERING - Creating business metrics
# ============================================
print("\n⚙️ Creating business metrics for analysis...")

# Convert date columns
df['issue_date'] = pd.to_datetime(df['issue_date'])
df['issue_month'] = df['issue_date'].dt.to_period('M')
df['issue_year'] = df['issue_date'].dt.year
df['issue_month_name'] = df['issue_date'].dt.month_name()

# Create loan status categories
df['good_bad_flag'] = np.where(
    df['loan_status'] == 'Fully Paid', 'Good Loan',
    np.where(df['loan_status'] == 'Charged Off', 'Bad Loan', 'Current')
)

# Create loan amount buckets
df['loan_amount_bucket'] = pd.cut(
    df['loan_amount'],
    bins=[0, 5000, 10000, 15000, 20000, 25000, 30000, 35000],
    labels=['0-5K', '5K-10K', '10K-15K', '15K-20K', '20K-25K', '25K-30K', '30K-35K']
)

# Create interest rate buckets
df['int_rate_bucket'] = pd.cut(
    df['int_rate'] * 100,
    bins=[0, 10, 15, 20, 25],
    labels=['0-10%', '10-15%', '15-20%', '20-25%']
)

# Create DTI buckets
df['dti_bucket'] = pd.cut(
    df['dti'] * 100,
    bins=[0, 10, 15, 20, 25, 30],
    labels=['0-10%', '10-15%', '15-20%', '20-25%', '25-30%']
)

# ============================================
# 4. CORE KPI CALCULATIONS
# ============================================
print("\n💰 Calculating key business metrics...")

# Total KPIs
total_applications = len(df)
total_funded = df['loan_amount'].sum()
total_received = df['total_payment'].sum()
avg_int_rate = df['int_rate'].mean() * 100
avg_dti = df['dti'].mean() * 100

print(f"📊 Total Loan Applications: {total_applications:,}")
print(f"💰 Total Funded Amount: ${total_funded:,.2f}")
print(f"💰 Total Received Amount: ${total_received:,.2f}")
print(f"📉 Average Interest Rate: {avg_int_rate:.2f}%")
print(f"📊 Average DTI: {avg_dti:.2f}%")

# ============================================
# 5. MTD vs PMTD CALCULATIONS
# ============================================
print("\n📅 Calculating Month-over-Month trends...")

# Get latest month
latest_date = df['issue_date'].max()
latest_year = latest_date.year
latest_month = latest_date.month

# Calculate previous month
if latest_month == 1:
    prev_year = latest_year - 1
    prev_month = 12
else:
    prev_year = latest_year
    prev_month = latest_month - 1

# Filter MTD and PMTD data
mtd_data = df[
    (df['issue_date'].dt.year == latest_year) & 
    (df['issue_date'].dt.month == latest_month)
]

pmtd_data = df[
    (df['issue_date'].dt.year == prev_year) & 
    (df['issue_date'].dt.month == prev_month)
]

# Calculate MTD metrics
mtd_apps = len(mtd_data)
mtd_funded = mtd_data['loan_amount'].sum()
mtd_received = mtd_data['total_payment'].sum()
mtd_int_rate = mtd_data['int_rate'].mean() * 100
mtd_dti = mtd_data['dti'].mean() * 100

# Calculate PMTD metrics
pmtd_apps = len(pmtd_data)
pmtd_funded = pmtd_data['loan_amount'].sum()
pmtd_received = pmtd_data['total_payment'].sum()
pmtd_int_rate = pmtd_data['int_rate'].mean() * 100
pmtd_dti = pmtd_data['dti'].mean() * 100

# Calculate MoM changes
mom_apps_pct = ((mtd_apps - pmtd_apps) / pmtd_apps * 100) if pmtd_apps > 0 else 0
mom_funded_pct = ((mtd_funded - pmtd_funded) / pmtd_funded * 100) if pmtd_funded > 0 else 0
mom_received_pct = ((mtd_received - pmtd_received) / pmtd_received * 100) if pmtd_received > 0 else 0

print(f"\n📈 MTD ({latest_date.strftime('%B %Y')}):")
print(f"   Applications: {mtd_apps:,} ({mom_apps_pct:+.1f}% MoM)")
print(f"   Funded: ${mtd_funded:,.2f} ({mom_funded_pct:+.1f}% MoM)")
print(f"   Received: ${mtd_received:,.2f} ({mom_received_pct:+.1f}% MoM)")

# ============================================
# 6. GOOD LOAN vs BAD LOAN ANALYSIS
# ============================================
print("\n✅❌ Good Loan vs Bad Loan Analysis:")

good_loans = df[df['loan_status'] == 'Fully Paid']
bad_loans = df[df['loan_status'] == 'Charged Off']

good_count = len(good_loans)
bad_count = len(bad_loans)
good_pct = (good_count / total_applications * 100)
bad_pct = (bad_count / total_applications * 100)

good_funded = good_loans['loan_amount'].sum()
bad_funded = bad_loans['loan_amount'].sum()
good_received = good_loans['total_payment'].sum()
bad_received = bad_loans['total_payment'].sum()

print(f"   ✅ Good Loans: {good_count:,} ({good_pct:.2f}%)")
print(f"   ❌ Bad Loans: {bad_count:,} ({bad_pct:.2f}%)")
print(f"   ✅ Good Loan Funded: ${good_funded:,.2f}")
print(f"   ❌ Bad Loan Funded: ${bad_funded:,.2f}")
print(f"   ✅ Good Loan Received: ${good_received:,.2f}")
print(f"   ❌ Bad Loan Received: ${bad_received:,.2f}")

# ============================================
# 7. SEGMENTATION ANALYSIS
# ============================================
print("\n📊 Segmentation Analysis:")

# By grade
grade_analysis = df.groupby('grade').agg({
    'id': 'count',
    'loan_amount': 'sum',
    'total_payment': 'sum',
    'int_rate': 'mean'
}).round(2)
print("\n📊 Loan Performance by Grade:")
print(grade_analysis)

# By home ownership
home_analysis = df.groupby('home_ownership').agg({
    'id': 'count',
    'loan_amount': 'sum',
    'total_payment': 'sum'
}).round(2)
print("\n🏠 Loan Performance by Home Ownership:")
print(home_analysis)

# By purpose
purpose_analysis = df.groupby('purpose').agg({
    'id': 'count',
    'loan_amount': 'sum',
    'total_payment': 'sum'
}).round(2)
print("\n🎯 Loan Performance by Purpose:")
print(purpose_analysis)

# By term
term_analysis = df.groupby('term').agg({
    'id': 'count',
    'loan_amount': 'sum',
    'total_payment': 'sum'
}).round(2)
print("\n⏱️ Loan Performance by Term:")
print(term_analysis)

# ============================================
# 8. MONTHLY TRENDS
# ============================================
print("\n📈 Monthly Trends:")

monthly_trends = df.groupby(df['issue_date'].dt.to_period('M')).agg({
    'id': 'count',
    'loan_amount': 'sum',
    'total_payment': 'sum'
}).reset_index()
monthly_trends.columns = ['Month', 'Applications', 'Funded', 'Received']

print(monthly_trends.to_string(index=False))

# ============================================
# 9. EXPORT CLEAN DATA
# ============================================
df.to_csv('bank_loan_clean.csv', index=False)
print("\n✅ Clean data exported for SQL/Power BI")

# Business Summary
print("\n" + "="*60)
print("🎯 BUSINESS SUMMARY (from Python)")
print("="*60)
print(f"Total Loan Applications: {total_applications:,}")
print(f"Total Funded Amount: ${total_funded:,.2f}")
print(f"Total Received Amount: ${total_received:,.2f}")
print(f"Average Interest Rate: {avg_int_rate:.2f}%")
print(f"Average DTI: {avg_dti:.2f}%")
print(f"Good Loan Rate: {good_pct:.2f}%")
print(f"Bad Loan Rate: {bad_pct:.2f}%")
print(f"Net Profit: ${total_received - total_funded:,.2f}")
print(f"ROI: {((total_received - total_funded) / total_funded * 100):.2f}%")
print("="*60)
```

```sql
/*
BANK LOAN ANALYSIS - SQL SERVER QUERIES
Business Goal: Answer specific business questions about loan portfolio
Focus: Portfolio health, risk analysis, performance metrics
*/

-- ============================================
-- 1. DATABASE SETUP - Loan Data Warehouse
-- ============================================
CREATE DATABASE BankLoanDB;
GO

USE BankLoanDB;
GO

-- Main loan applications table
CREATE TABLE LoanApplications (
    LoanID INT PRIMARY KEY,
    MemberID INT,
    LoanStatus VARCHAR(50),
    Grade CHAR(1),
    SubGrade VARCHAR(2),
    Term VARCHAR(20),
    HomeOwnership VARCHAR(20),
    EmpLength VARCHAR(20),
    Purpose VARCHAR(50),
    IssueDate DATE,
    LastCreditPullDate DATE,
    LastPaymentDate DATE,
    NextPaymentDate DATE,
    AnnualIncome DECIMAL(12,2),
    DTI DECIMAL(5,4),
    Installment DECIMAL(10,2),
    IntRate DECIMAL(5,4),
    LoanAmount DECIMAL(10,2),
    TotalPayment DECIMAL(12,2),
    TotalAcc INT,
    AddressState VARCHAR(2),
    ApplicationType VARCHAR(20),
    VerificationStatus VARCHAR(20),
    CreatedDate DATETIME DEFAULT GETDATE()
);

-- Load data (from Python cleaned file)
BULK INSERT LoanApplications
FROM 'C:\data\bank_loan_clean.csv'
WITH (
    FIRSTROW = 2,
    FIELDTERMINATOR = ',',
    ROWTERMINATOR = '\n',
    TABLOCK
);

-- ============================================
-- 2. BUSINESS QUESTION: Overall Portfolio Summary
-- ============================================
SELECT 
    'Total Loan Applications' AS Metric,
    FORMAT(COUNT(*), 'N0') AS Value
FROM LoanApplications
UNION ALL
SELECT 'Total Funded Amount',
    CONCAT('$', FORMAT(SUM(LoanAmount), 'N0'))
FROM LoanApplications
UNION ALL
SELECT 'Total Received Amount',
    CONCAT('$', FORMAT(SUM(TotalPayment), 'N0'))
FROM LoanApplications
UNION ALL
SELECT 'Average Interest Rate',
    CONCAT(FORMAT(AVG(IntRate) * 100, 'N2'), '%')
FROM LoanApplications
UNION ALL
SELECT 'Average DTI',
    CONCAT(FORMAT(AVG(DTI) * 100, 'N2'), '%')
FROM LoanApplications
UNION ALL
SELECT 'Good Loan %',
    CONCAT(FORMAT(
        100.0 * SUM(CASE WHEN LoanStatus = 'Fully Paid' THEN 1 ELSE 0 END) / COUNT(*), 
        'N2'), '%')
FROM LoanApplications
UNION ALL
SELECT 'Bad Loan %',
    CONCAT(FORMAT(
        100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*), 
        'N2'), '%')
FROM LoanApplications;

-- ============================================
-- 3. BUSINESS QUESTION: MTD vs PMTD Performance
-- ============================================
DECLARE @LatestDate DATE;
DECLARE @LatestYear INT;
DECLARE @LatestMonth INT;
DECLARE @PrevYear INT;
DECLARE @PrevMonth INT;

-- Get latest month
SELECT @LatestDate = MAX(IssueDate) FROM LoanApplications;
SET @LatestYear = YEAR(@LatestDate);
SET @LatestMonth = MONTH(@LatestDate);

-- Calculate previous month
IF @LatestMonth = 1
BEGIN
    SET @PrevYear = @LatestYear - 1;
    SET @PrevMonth = 12;
END
ELSE
BEGIN
    SET @PrevYear = @LatestYear;
    SET @PrevMonth = @LatestMonth - 1;
END;

-- MTD Metrics
SELECT 
    'MTD' AS Period,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI
FROM LoanApplications
WHERE YEAR(IssueDate) = @LatestYear AND MONTH(IssueDate) = @LatestMonth

UNION ALL

-- PMTD Metrics
SELECT 
    'PMTD' AS Period,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI
FROM LoanApplications
WHERE YEAR(IssueDate) = @PrevYear AND MONTH(IssueDate) = @PrevMonth;

-- ============================================
-- 4. BUSINESS QUESTION: Good Loan vs Bad Loan Analysis
-- ============================================
WITH LoanCategories AS (
    SELECT 
        CASE 
            WHEN LoanStatus = 'Fully Paid' THEN 'Good Loan'
            WHEN LoanStatus = 'Charged Off' THEN 'Bad Loan'
            ELSE 'Current'
        END AS LoanCategory,
        LoanAmount,
        TotalPayment,
        IntRate,
        DTI
    FROM LoanApplications
)
SELECT 
    LoanCategory,
    COUNT(*) AS ApplicationCount,
    FORMAT(100.0 * COUNT(*) / SUM(COUNT(*)) OVER(), 'N2') AS Percentage,
    SUM(LoanAmount) AS TotalFunded,
    SUM(TotalPayment) AS TotalReceived,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI,
    -- Profit/Loss
    SUM(TotalPayment) - SUM(LoanAmount) AS NetProfit,
    CASE 
        WHEN SUM(LoanAmount) > 0 
        THEN (SUM(TotalPayment) - SUM(LoanAmount)) / SUM(LoanAmount) * 100
        ELSE 0
    END AS ROI
FROM LoanCategories
GROUP BY LoanCategory
ORDER BY 
    CASE LoanCategory
        WHEN 'Good Loan' THEN 1
        WHEN 'Current' THEN 2
        WHEN 'Bad Loan' THEN 3
    END;

-- ============================================
-- 5. BUSINESS QUESTION: Monthly Trend Analysis
-- ============================================
SELECT 
    YEAR(IssueDate) AS Year,
    MONTH(IssueDate) AS Month,
    DATENAME(month, IssueDate) AS MonthName,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI,
    -- Good loan rate
    100.0 * SUM(CASE WHEN LoanStatus = 'Fully Paid' THEN 1 ELSE 0 END) / COUNT(*) AS GoodLoanRate,
    -- Month-over-month growth
    LAG(SUM(LoanAmount)) OVER (ORDER BY YEAR(IssueDate), MONTH(IssueDate)) AS PrevMonthFunded,
    (SUM(LoanAmount) - LAG(SUM(LoanAmount)) OVER (ORDER BY YEAR(IssueDate), MONTH(IssueDate))) / 
        NULLIF(LAG(SUM(LoanAmount)) OVER (ORDER BY YEAR(IssueDate), MONTH(IssueDate)), 0) * 100 AS MoMGrowth
FROM LoanApplications
GROUP BY YEAR(IssueDate), MONTH(IssueDate), DATENAME(month, IssueDate)
ORDER BY Year, Month;

-- ============================================
-- 6. BUSINESS QUESTION: Regional Analysis by State
-- ============================================
SELECT 
    AddressState AS State,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(LoanAmount) AS AvgLoanAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    -- Good loan rate by state
    100.0 * SUM(CASE WHEN LoanStatus = 'Fully Paid' THEN 1 ELSE 0 END) / COUNT(*) AS GoodLoanRate,
    -- Risk ranking
    RANK() OVER (ORDER BY 
        100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*) DESC) AS RiskRank
FROM LoanApplications
GROUP BY AddressState
HAVING COUNT(*) > 100  -- Only states with significant volume
ORDER BY FundedAmount DESC;

-- ============================================
-- 7. BUSINESS QUESTION: Loan Term Analysis
-- ============================================
SELECT 
    Term,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(LoanAmount) AS AvgLoanAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI,
    -- Performance metrics
    100.0 * SUM(CASE WHEN LoanStatus = 'Fully Paid' THEN 1 ELSE 0 END) / COUNT(*) AS GoodLoanRate,
    100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*) AS BadLoanRate,
    -- Profitability
    SUM(TotalPayment) - SUM(LoanAmount) AS NetProfit
FROM LoanApplications
GROUP BY Term
ORDER BY Term;

-- ============================================
-- 8. BUSINESS QUESTION: Grade Analysis
-- ============================================
SELECT 
    Grade,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(LoanAmount) AS AvgLoanAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI,
    -- Risk by grade
    100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*) AS DefaultRate,
    -- Recommendation
    CASE 
        WHEN 100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*) > 20 THEN 'HIGH RISK - REVIEW'
        WHEN 100.0 * SUM(CASE WHEN LoanStatus = 'Charged Off' THEN 1 ELSE 0 END) / COUNT(*) > 15 THEN 'MEDIUM RISK - MONITOR'
        ELSE 'LOW RISK - APPROVE'
    END AS RiskLevel
FROM LoanApplications
GROUP BY Grade
ORDER BY Grade;

-- ============================================
-- 9. BUSINESS QUESTION: Purpose Analysis
-- ============================================
SELECT 
    Purpose,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(LoanAmount) AS AvgLoanAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    -- Success rate
    100.0 * SUM(CASE WHEN LoanStatus = 'Fully Paid' THEN 1 ELSE 0 END) / COUNT(*) AS SuccessRate,
    -- Ranking
    RANK() OVER (ORDER BY SUM(LoanAmount) DESC) AS VolumeRank
FROM LoanApplications
GROUP BY Purpose
ORDER BY FundedAmount DESC;

-- ============================================
-- 10. BUSINESS QUESTION: Home Ownership Analysis
-- ============================================
SELECT 
    HomeOwnership,
    COUNT(*) AS Applications,
    SUM(LoanAmount) AS FundedAmount,
    SUM(TotalPayment) AS ReceivedAmount,
    AVG(LoanAmount) AS AvgLoanAmount,
    AVG(IntRate) * 100 AS AvgIntRate,
    AVG(DTI) * 100 AS AvgDTI,
    -- Risk assessment
    CASE 
        WHEN AVG(DTI) * 100 > 20 THEN 'HIGH RISK'
        WHEN AVG(DTI) * 100 > 15 THEN 'MEDIUM RISK'
        ELSE 'LOW RISK'
    END AS RiskCategory
FROM LoanApplications
GROUP BY HomeOwnership
ORDER BY FundedAmount DESC;

-- ============================================
-- 11. STORED PROCEDURE: Portfolio Health Report
-- ============================================
CREATE PROCEDURE sp_GetPortfolioHealth
    @AsOfDate DATE = NULL
AS
BEGIN
    IF @AsOfDate IS NULL
        SET @AsOfDate = GETDATE();

    SELECT
        -- Overall Metrics
        (SELECT COUNT(*) FROM LoanApplications WHERE IssueDate <= @AsOfDate) AS TotalApplications,
        (SELECT SUM(LoanAmount) FROM LoanApplications WHERE IssueDate <= @AsOfDate) AS TotalFunded,
        (SELECT SUM(TotalPayment) FROM LoanApplications WHERE IssueDate <= @AsOfDate) AS TotalReceived,
        
        -- Current Month Metrics
        (SELECT COUNT(*) FROM LoanApplications 
         WHERE YEAR(IssueDate) = YEAR(@AsOfDate) AND MONTH(IssueDate) = MONTH(@AsOfDate)) AS MTDApplications,
        (SELECT SUM(LoanAmount) FROM LoanApplications 
         WHERE YEAR(IssueDate) = YEAR(@AsOfDate) AND MONTH(IssueDate) = MONTH(@AsOfDate)) AS MTDFunded,
        
        -- Portfolio Quality
        (SELECT 100.0 * COUNT(*) / (SELECT COUNT(*) FROM LoanApplications)
         FROM LoanApplications WHERE LoanStatus = 'Fully Paid') AS GoodLoanRate,
        (SELECT 100.0 * COUNT(*) / (SELECT COUNT(*) FROM LoanApplications)
         FROM LoanApplications WHERE LoanStatus = 'Charged Off') AS BadLoanRate,
        
        -- Risk Metrics
        (SELECT AVG(DTI) * 100 FROM LoanApplications) AS AvgDTI,
        (SELECT AVG(IntRate) * 100 FROM LoanApplications) AS AvgIntRate,
        
        -- Top Risk States
        (SELECT TOP 1 AddressState FROM LoanApplications 
         GROUP BY AddressState 
         ORDER BY AVG(CASE WHEN LoanStatus = 'Charged Off' THEN 1.0 ELSE 0 END) DESC) AS HighestRiskState;
END;
GO
```

```python
"""
BANK LOAN ANALYSIS - AWS IMPLEMENTATION
Business Goal: Enable team-wide access to loan data, run ad-hoc queries

import boto3
import pandas as pd
from io import StringIO
import time

# ============================================
# 1. AWS S3 SETUP - Central Data Lake
# ============================================
print("🚀 Setting up AWS S3 data lake for loan data...")

# Configure AWS (using credentials from environment)
s3_client = boto3.client(

    region_name='us-east-1'  # Virginia - good for cost

# Create bucket for loan data
bucket_name = 'bank-loan-analytics-2026'
try:
    s3_client.create_bucket(
        Bucket=bucket_name
    )
    print(f"✅ Created bucket: {bucket_name}")
except:
    print(f"ℹ️ Bucket {bucket_name} already exists")

# ============================================
# 2. ORGANIZE DATA IN S3 (Like folders for business units)
# ============================================
print("\n📁 Organizing loan data in S3...")

# Structure: s3://bucket/category/year/month/file.csv
folders = [
    'raw/loans/2021/',
    'raw/loans/2022/',
    'processed/daily/',
    'processed/monthly/',
    'reports/portfolio/',
    'reports/risk/',
    'reports/executive/',
    'athena_results/'
]

for folder in folders:
    s3_client.put_object(
        Bucket=bucket_name,
        Key=folder
    )
print("✅ Folder structure created")

# ============================================
# 3. UPLOAD DATA TO S3
# ============================================
print("\n📤 Uploading loan data to S3...")

# Read cleaned data from Python
df = pd.read_csv('bank_loan_clean.csv')

# Convert to CSV in memory
csv_buffer = StringIO()
df.to_csv(csv_buffer, index=False)

# Upload raw data
s3_client.put_object(
    Bucket=bucket_name,
    Key='raw/loans/2021/loan_data_2021.csv',
    Body=csv_buffer.getvalue()
)

# Also upload partitioned by grade (for faster queries)
for grade in df['grade'].unique():
    grade_df = df[df['grade'] == grade]
    grade_buffer = StringIO()
    grade_df.to_csv(grade_buffer, index=False)
    
    s3_client.put_object(
        Bucket=bucket_name,
        Key=f'processed/daily/grade={grade}/loans.csv',
        Body=grade_buffer.getvalue()
    )
print(f"✅ Uploaded {len(df):,} loan records to S3")

# ============================================
# 4. AWS ATHENA SETUP - Serverless SQL Queries
# ============================================
print("\n🔍 Setting up AWS Athena for serverless queries...")

athena_client = boto3.client('athena', region_name='us-east-1')

# Create database in Athena
create_db_query = "CREATE DATABASE IF NOT EXISTS bank_loan_analytics;"

response = athena_client.start_query_execution(
    QueryString=create_db_query,
    QueryExecutionContext={'Database': 'bank_loan_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}
)

print("✅ Athena database created")

# ============================================
# 5. CREATE EXTERNAL TABLE IN ATHENA
# ============================================
print("\n📊 Creating external table for loan data...")

create_table_query = f"""
CREATE EXTERNAL TABLE IF NOT EXISTS loans (
    loan_id INT,
    member_id INT,
    loan_status STRING,
    grade STRING,
    sub_grade STRING,

    home_ownership STRING,

ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'

LOCATION 's3://{bucket_name}/raw/loans/'
TBLPROPERTIES ('has_encrypted_data'='false');

response = athena_client.start_query_execution(
    QueryString=create_table_query,
    QueryExecutionContext={'Database': 'bank_loan_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena_results/'}

# Wait for query to complete
execution_id = response['QueryExecutionId']

query1 = """

query2 = """
```
